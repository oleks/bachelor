\frametitle{Observed mistakes}

\begin{itemize}

\item List indexing

\begin{itemize}

\item Lists are sometimes 0-indexed rather than 1-indexed.

\item $\forall\ \{i\mid 0\geq i < |P|\}$ should obviously be $\forall\ \{i\mid
0< i \leq |P|\}$.

\end{itemize}

\end{itemize}
